207 /* @brief Fast/efficient version of #PIN_setOutputEnable()
208  * @note Does not include any checks on handle for efficiency reasons,
209  * use #PIN_setOutputEnable() for checked version
210  */
211 __STATIC_INLINE void PINCC26XX_setOutputEnable(PIN_Id pinId, bool outputEnable) {
212  uint32_t key = HwiP_disable();
213  HWREG(GPIO_BASE + GPIO_O_DOE31_0) =
214  ((HWREG(GPIO_BASE + GPIO_O_DOE31_0) & ~(1 << pinId)) | (outputEnable << pinId));
215  HwiP_restore(key);
216 }

268 __STATIC_INLINE void PINCC26XX_setPortOutputEnable(PIN_Handle handle, uint32_t outputEnableMask) {
269  // Only a single port on CC26xx
270  uint32_t key = HwiP_disable();
271  HWREG(GPIO_BASE + GPIO_O_DOE31_0) =
272  (HWREG(GPIO_BASE + GPIO_O_DOE31_0) & (~handle->portMask)) | (outputEnableMask & handle->portMask);
273  HwiP_restore(key);
